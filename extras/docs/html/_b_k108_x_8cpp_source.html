140 /**
141  * @ingroup GA02
142  * @brief Sends a Byte to the slave device
143  * @param data to be sent to the slave device
144  */
145 void BK108X::i2cWriteByte( uint8_t data)
146 {
147  pinMode(pin_sdio, OUTPUT);
148  delayMicroseconds(1);
149 
150  for (int i = 0; i < 8; i++) {
151  digitalWrite(this->pin_sdio, (data & this->deviceAddress) ); // SDA will be 1 or 0
152  delayMicroseconds(1);
153  digitalWrite(this->pin_sclk, HIGH);
154  delayMicroseconds(1);
155  data = data << 1;
156  digitalWrite(this->pin_sclk, LOW);
157  }
158 }

162  * @brief Gets a Byte from the slave device
163  * @return value read from the device
164  */
165 uint8_t BK108X::i2cReadByte()
166 {
167  uint8_t value = 0;
168 
169  pinMode(pin_sdio, INPUT);
170  delayMicroseconds(1);
171 
172  for (int i = 0; i < 8; i++)
173  {
174  digitalWrite(this->pin_sclk, HIGH);
175  value = value << 1;
176  delayMicroseconds(1);
177  if ( digitalRead(this->pin_sdio) )
178  value = value | 1;
179  digitalWrite(this->pin_sclk, LOW);
180  delayMicroseconds(1);
181  }
182 
183  return value;
184 }

193 void BK108X::writeRegister(uint8_t reg, uint8_t *data, uint8_t size) {
194  this->i2cStart();
195  this->i2cWriteByte(this->deviceAddress);
196  this->i2cReceiveAck();
197 
198  // reg = (reg << 1) | 1;
199  reg = reg << 1;
200 
201  this->i2cWriteByte(reg);
202  this->i2cReceiveAck();
203 
204  for (int i = 0; i < size; i++)
205  {
206  this->i2cWriteByte(data[i]);
207  this->i2cReceiveAck();
208  }
209  this->i2cStop();
210 }

212 /**
213  * @ingroup GA02
214  * @brief Gets an array of values from a BK108X given register
215  * @param reg register to be read
216  * @param data byte array of values to be populated
217  * @param size size of byte array
218  */
219 void BK108X::readRegister(uint8_t reg, uint8_t *data, uint8_t size) {
220  this->i2cStart();
221  this->i2cWriteByte(this->deviceAddress);
222  this->i2cReceiveAck();
223 
224  reg = (reg << 1) | 1;
225 
226  this->i2cWriteByte(reg);
227  this->i2cReceiveAck();
228 
229  for (int i = 0; i < size - 1; i++ ) {
230  data[i] = this->i2cReadByte();
231  this->i2cAck();
232  }
233  data[size - 1] = this->i2cReadByte();
234  this->i2cNack();
235  this->i2cStop();
236 }

264 /**
265  * @ingroup GA03
266  * @brief Sets a given value to the device registers
267  * @details For write operations, the device acknowledge is followed by an eight bit data word latched internally on rising edges of SCLK. The device acknowledges each byte of data written by driving SDIO low after the next falling SCLK edge, for 1 cycle.
268  * @details An internal address counter automatically increments to allow continuous data byte writes, starting with the upper byte of register 02h, followed by the lower byte of register 02h, and onward until the lower byte of the last register is reached. The internal address counter then automatically wraps around to the upper byte of register 00h and proceeds from there until continuous writes end.
269  * @details The registers from 0x2 to 0x07 are used to setup the device. This method writes the array shadowRegisters, elements 8 to 14 (corresponding the registers 0x2 to 0x7 respectively) into the device. See Device registers map in BK108X.h file.
270  * @details To implement this, a register maping was created to deal with each register structure. For each type of register, there is a reference to the array element.
271  *
272  * @see shadowRegisters;
273  *
274  * @param device register address
275  */
276 void BK108X::setRegister(uint8_t reg, uint16_t value)
277 {
278  word16_to_bytes param;
279 
280  param.raw = value;
281 
282  this->writeRegister(reg, param.array,2);
283 
284  shadowRegisters[reg] = value; // Syncs with the shadowRegisters
285 }
